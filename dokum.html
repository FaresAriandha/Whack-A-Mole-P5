void upAndDownShow() {
  if (naik) {
    mRandom.transisiDown(numberOfLevel);
  } else {
    if (
      mouseX >= mRandom.xPos - mRandom.wSize / 2 &&
      mouseX <= mRandom.xPos + mRandom.wSize / 2 &&
      mouseY >= mRandom.yPos - mRandom.hSize / 2 &&
      mouseY <= mRandom.yPos + mRandom.hSize / 2 &&
      mousePressed
    ) {
      naik = true;
      bsPukul.play();
      skor++;
    } else {
      mRandom.transisiMuncul(numberOfLevel);
    }
  }
}

void gamePlay() {
  cursor(ARROW);
  if (waktuTunggu > -1) {
    rundownBeforePlay();
    if (waktuTunggu == 0) {
      countDown = 0;
      startTime = millis()/1000;
      playingTime();
    }
  } else {
    if (a) {
      skor = 0;
      randomPosition();
      a = false;
    }
    if(countDown > 0){
      playingTime();
    }
    mainPlay();
    if (countDown <= 0) {
      winOrLose();
      finish = true;
    } else {
      changeCursor();
    }
  }
}
